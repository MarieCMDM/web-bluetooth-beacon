import socket print("Waiting for connections") scontrol = socket.socket( socket.AF_BLUETOOTH, socket.SOCK_SEQPACKET, socket.BTPROTO_L2CAP) # BluetoothSocket(L2CAP) sinterrupt = socket.socket( socket.AF_BLUETOOTH, socket.SOCK_SEQPACKET, socket.BTPROTO_L2CAP) # BluetoothSocket(L2CAP) scontrol.setsockopt(socket.SOL_SOCKET, socket.SO_REUSEADDR, 1) sinterrupt.setsockopt(socket.SOL_SOCKET, socket.SO_REUSEADDR, 1) # bind these sockets to a port - port zero to select next available scontrol.bind((socket.BDADDR_ANY, self.P_CTRL)) sinterrupt.bind((socket.BDADDR_ANY, self.P_INTR)) # Start listening on the server sockets scontrol.listen(5) # Limit of 1 connection sinterrupt.listen(5) ccontrol, cinfo = scontrol.accept() print ("Got a connection on the control channel from " + cinfo[0]) cinterrupt, cinfo = sinterrupt.accept() print ("Got a connection on the interrupt channel from " + cinfo[0]) # send a string to the bluetooth host machine
